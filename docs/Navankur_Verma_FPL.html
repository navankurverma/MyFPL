Motivation¶

Each year approx 7 million users sign up and create their fantasy football team. Making it most widely used platform for its recreational aspect and now due to its popularity it has turned into a business. I have been using it myself for last 4 years with my friends and colleagues. Alongwith fantasy data, it also holds data of English Premier League data sets players and teams data

def fetch_myteam_gw(teamID, gw):
    url = 'https://fantasy.premierleague.com/api/entry/{}/event/{}/picks/'.format(teamID,gw)
    team = requests.get(url)
    return team.json()

gameweeks = []
for x in fpl['events']:
    if 'chip_plays' in x:
        x['bboost#'] = 0
        x['3xc#'] = 0
        x['wildcard#'] = 0
        x['freehit#'] = 0
        for y in x['chip_plays']:
            if (y['chip_name']=='bboost'):
                x['bboost#'] = y['num_played']
            if (y['chip_name']=='3xc'):
                x['3xc#'] = y['num_played']
            if (y['chip_name']=='wildcard'):
                x['wildcard#'] = y['num_played']
            if (y['chip_name']=='freehit'):
                x['freehit#'] = y['num_played']
    if 'top_element_info' in x:
        if (x['top_element_info'] == None):
            x['top_element_points'] = 0
        else:
            x['top_element_points'] = x['top_element_info']['points']
    gameweeks.append(x)
gameweeks_details = pd.DataFrame(gameweeks)

def fetch_player_hist(player_id):
    url = 'https://fantasy.premierleague.com/api/element-summary/{}/'.format(player_id)
    plyr = requests.get(url)
    return plyr.json()
